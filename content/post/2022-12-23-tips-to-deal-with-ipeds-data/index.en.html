library(tidyverse)

Merge multiple state-level dataframes by state and year

merge_cols <- c("State","Year")
df <- Reduce(function(x,y) merge(x,y, by = merge_cols), list(df1, df2, df3, df4,...))

Replace state full name with state abbreviations

x <- unique(DF$State)
DF[[i]]$State <- state.abb[match(x,state.name)]
DF$State <- state.abb[match(DF$state,state.name)]
